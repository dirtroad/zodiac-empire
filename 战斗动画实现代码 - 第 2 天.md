## 一、攻击动画（前摇/冲锋/返回）

### 文件：`client/assets/scripts/scenes/BattleSceneController.ts`

```typescript
import { _decorator, Component, Node, Vec3, tween, UITransform, Color } from 'cc';
const { ccclass, property } = _decorator;

@ccclass('BattleSceneController')
export class BattleSceneController extends Component {
    
    @property(Node)
    attackerNode: Node = null; // 攻击方
    
    @property(Node)
    defenderNode: Node = null; // 防御方
    
    // 攻击动画序列
    async playAttackAnimation(skillType: string = 'attack') {
        console.log('[战斗动画] 开始攻击动画:', skillType);
        
        const attackerPos = this.attackerNode.getPosition();
        const defenderPos = this.defenderNode.getPosition();
        
        // 1. 前摇动画（蓄力 0.3 秒）
        await this.playWindupAnimation(attackerPos);
        
        // 2. 冲锋动画（冲向对方 0.2 秒）
        await this.playChargeAnimation(attackerPos, defenderPos);
        
        // 3. 显示技能特效
        this.showSkillEffect(skillType, defenderPos);
        
        // 4. 受击方震动
        this.playHitAnimation();
        
        // 5. 返回原位（0.2 秒）
        await this.playReturnAnimation(attackerPos, defenderPos);
        
        console.log('[战斗动画] 攻击动画完成');
    }
    
    // 1. 前摇动画（蓄力）
    async playWindupAnimation(originalPos: Vec3) {
        return new Promise<void>((resolve) => {
            tween(this.attackerNode)
                .to(0.1, { scale: new Vec3(1.3, 1.3, 1) }) // 放大
                .to(0.1, { scale: new Vec3(1.0, 1.0, 1) }) // 恢复
                .to(0.1, { scale: new Vec3(1.3, 1.3, 1) }) // 再放大
                .start();
            
            setTimeout(() => resolve(), 300);
        });
    }
    
    // 2. 冲锋动画（冲向对方）
    async playChargeAnimation(startPos: Vec3, endPos: Vec3) {
        return new Promise<void>((resolve) => {
            // 计算中间位置（向前冲锋）
            const midPos = new Vec3(
                startPos.x + (endPos.x - startPos.x) * 0.8,
                startPos.y,
                startPos.z
            );
            
            tween(this.attackerNode)
                .to(0.1, { position: midPos }) // 快速冲锋
                .start();
            
            setTimeout(() => resolve(), 200);
        });
    }
    
    // 3. 显示技能特效
    showSkillEffect(skillType: string, targetPos: Vec3) {
        const emojiMap: Record<string, string> = {
            'attack': '⚔️',
            'fire': '🔥',
            'water': '💧',
            'wind': '🌪️',
            'earth': '🪨',
            'light': '✨',
            'dark': '🌑',
            'defend': '🛡️',
        };
        
        const emoji = emojiMap[skillType] || '⚔️';
        this.createFloatingText(emoji, targetPos, 60, true);
    }
    
    // 4. 受击动画（震动）
    async playHitAnimation() {
        return new Promise<void>((resolve) => {
            tween(this.defenderNode)
                .by(0.05, { position: new Vec3(-10, 0, 0) }) // 左移
                .by(0.05, { position: new Vec3(10, 0, 0) })  // 右移
                .by(0.05, { position: new Vec3(-10, 0, 0) }) // 左移
                .by(0.05, { position: new Vec3(10, 0, 0) })  // 右移
                .start();
            
            setTimeout(() => resolve(), 200);
        });
    }
    
    // 5. 返回动画（回到原位）
    async playReturnAnimation(originalPos: Vec3, currentPos: Vec3) {
        return new Promise<void>((resolve) => {
            tween(this.attackerNode)
                .to(0.2, { position: originalPos }) // 缓慢返回
                .start();
            
            setTimeout(() => resolve(), 200);
        });
    }
}
```

## 二、伤害飘字动画

### 文件：`client/assets/scripts/ui/battle/BattleUI.ts`

```typescript
import { _decorator, Component, Node, Label, Color, Vec3, tween, UITransform } from 'cc';
const { ccclass, property } = _decorator;

@ccclass('BattleUI')
export class BattleUI extends Component {
    
    @property(Node)
    damageTextPrefab: Node = null; // 伤害文字预制体
    
    @property(Node)
    battleField: Node = null; // 战斗区域父节点
    
    // 显示伤害飘字
    showDamage(value: number, isCrit: boolean, target: Node) {
        const text = isCrit ? `💥${value}` : `-${value}`;
        const color = isCrit ? new Color(255, 50, 50) : new Color(255, 255, 255); // 暴击红色，普通白色
        
        this.createFloatingText(text, target.getPosition(), 40, false, color, isCrit);
    }
    
    // 创建飘字
    createFloatingText(
        text: string, 
        startPos: Vec3, 
        fontSize: number = 40, 
        isEmoji: boolean = false,
        color: Color = new Color(255, 255, 255),
        isCrit: boolean = false
    ) {
        // 创建文字节点
        const textNode = new Node('DamageText');
        const label = textNode.addComponent(Label);
        
        label.string = text;
        label.fontSize = fontSize;
        label.color = color;
        
        // 如果是 emoji，使用系统字体
        if (isEmoji) {
            label.useSystemFont = true;
        }
        
        // 设置初始位置
        textNode.setPosition(startPos.x, startPos.y + 50, startPos.z);
        
        // 添加到战斗区域
        if (this.battleField) {
            this.battleField.addChild(textNode);
        } else {
            this.node.addChild(textNode);
        }
        
        // 播放飘字动画
        this.playFloatingAnimation(textNode, isCrit);
        
        // 2 秒后销毁
        setTimeout(() => {
            textNode.destroy();
        }, 2000);
        
        return textNode;
    }
    
    // 飘字动画
    playFloatingAnimation(textNode: Node, isCrit: boolean) {
        const duration = isCrit ? 1.5 : 1.0; // 暴击飘字时间更长
        const startY = textNode.getPosition().y;
        
        tween(textNode)
            .to(duration, { 
                position: new Vec3(
                    textNode.getPosition().x,
                    startY + 100, // 向上飘 100 像素
                    textNode.getPosition().z
                )
            })
            .by(0.1, { scale: new Vec3(0.2, 0.2, 1) }) // 稍微放大
            .to(0.5, { opacity: 0 }) // 最后淡出
            .start();
    }
    
    // 显示治疗飘字（绿色）
    showHeal(value: number, target: Node) {
        const text = `+${value}`;
        const color = new Color(100, 255, 100); // 绿色
        this.createFloatingText(text, target.getPosition(), 40, false, color);
    }
    
    // 显示状态文字（暴击/闪避/格挡）
    showStatusText(status: string, target: Node) {
        const emojiMap: Record<string, string> = {
            'crit': '💥暴击!',
            'dodge': '💨闪避!',
            'block': '🛡️格挡!',
            'miss': '❌未命中!',
        };
        
        const text = emojiMap[status] || status;
        this.createFloatingText(text, target.getPosition(), 50, true, new Color(255, 255, 100), true);
    }
}
```

```typescript
// 受击震动（带血量条闪烁）
playHitEffect(damage: number, isCrit: boolean) {
    // 1. 角色震动
    this.playHitAnimation();
    
    // 2. 血量条闪烁红色
    this.flashHPBar(isCrit);
    
    // 3. 显示伤害数字
    if (this.battleUI) {
        this.battleUI.showDamage(damage, isCrit, this.defenderNode);
    }
    
    // 4. 屏幕震动（可选）
    if (isCrit) {
        this.shakeCamera();
    }
}

// 血量条闪烁
async flashHPBar(isCrit: boolean) {
    const hpBar = this.hpBarNode; // 血量条节点
    const originalColor = hpBar.color;
    const flashColor = isCrit ? new Color(255, 0, 0) : new Color(255, 100, 100); // 暴击深红，普通浅红
    
    // 闪烁 3 次
    for (let i = 0; i < 3; i++) {
        hpBar.color = flashColor;
        await this.wait(100);
        hpBar.color = originalColor;
        await this.wait(100);
    }
}

// 屏幕震动（暴击时）
async shakeCamera() {
    const cameraNode = this.cameraNode; // 摄像机节点
    const originalPos = cameraNode.getPosition();
    
    // 快速震动 5 次
    for (let i = 0; i < 5; i++) {
        const offsetX = (Math.random() - 0.5) * 20;
        const offsetY = (Math.random() - 0.5) * 20;
        
        cameraNode.setPosition(
            originalPos.x + offsetX,
            originalPos.y + offsetY,
            originalPos.z
        );
        
        await this.wait(50);
    }
    
    // 恢复原位
    cameraNode.setPosition(originalPos);
}

// 等待工具函数
wait(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
}
```

## 四、完整战斗流程示例

### 文件：`client/assets/scripts/scenes/BattleSceneController.ts`

```typescript
// 完整战斗回合示例
async executeBattleRound() {
    console.log('[战斗] 第', this.currentRound, '回合开始');
    
    // 1. 攻击方行动
    await this.playAttackAnimation('fire'); // 火系技能
    
    // 2. 计算伤害
    const damage = this.calculateDamage();
    const isCrit = this.checkCrit();
    
    // 3. 受击方承受伤害
    this.defenderHP -= damage;
    this.playHitEffect(damage, isCrit);
    
    // 4. 更新血量条
    this.updateHPBar(this.defenderHP, this.defenderMaxHP);
    
    // 5. 检查是否死亡
    if (this.defenderHP <= 0) {
        this.onBattleEnd('victory');
        return;
    }
    
    // 6. 防御方反击（如果还活着）
    await this.wait(500); // 等待 0.5 秒
    await this.playAttackAnimation('attack'); // 普通攻击
    
    // 7. 计算反击伤害
    const counterDamage = this.calculateDamage();
    const counterCrit = this.checkCrit();
    
    // 8. 攻击方承受伤害
    this.attackerHP -= counterDamage;
    this.playHitEffect(counterDamage, counterCrit);
    this.updateHPBar(this.attackerHP, this.attackerMaxHP);
    
    // 9. 检查是否死亡
    if (this.attackerHP <= 0) {
        this.onBattleEnd('defeat');
        return;
    }
    
    // 10. 回合结束
    this.currentRound++;
    console.log('[战斗] 第', this.currentRound - 1, '回合结束');
}

// 伤害计算（含运气成分）
calculateDamage(): number {
    const baseDamage = this.attacker.attack;
    const defense = this.defender.defense;
    
    // 五行克制系数
    const wuxingCoeff = this.getWuxingCoefficient();
    
    // 幸运系数（80%-120%）
    const luckCoeff = 0.8 + Math.random() * 0.4;
    
    // 最终伤害
    let damage = baseDamage * wuxingCoeff * luckCoeff;
    
    // 防御减伤
    const defenseReduction = defense / (defense + 1000);
    damage = damage * (1 - defenseReduction);
    
    return Math.floor(damage);
}

// 暴击判定
checkCrit(): boolean {
    const critRate = this.attacker.critRate || 0.1; // 默认 10% 暴击率
    return Math.random() < critRate;
}

// 五行克制系数
getWuxingCoefficient(): number {
    // 简化版：相克 1.3，相生 0.9，同行 1.0
    const attackerWuxing = this.attacker.wuxing;
    const defenderWuxing = this.defender.wuxing;
    
    const wuxingMap: Record<string, string> = {
        'wood': 'fire',
        'fire': 'earth',
        'earth': 'metal',
        'metal': 'water',
        'water': 'wood',
    };
    
    if (wuxingMap[attackerWuxing] === defenderWuxing) {
        return 1.3; // 相克
    } else if (wuxingMap[defenderWuxing] === attackerWuxing) {
        return 0.9; // 被克
    } else {
        return 1.0; // 同行
    }
}
```
