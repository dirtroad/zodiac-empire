### 代码示例

#### 更新血量条
```typescript
// 玩家受到 5000 伤害
const newHP = 15000 - 5000;  // 10000
battleUI.updateHPBar('player', newHP, 15000);
// 血量条自动变为绿色（66.7% > 50%）

// 玩家再受到 6000 伤害
```

### 技术要点

#### 1. 血量条颜色逻辑
```typescript
const percent = currentHP / maxHP;

if (percent > 0.5) {
    color = Color.GREEN;   // 🟢 安全
} else if (percent > 0.3) {
    color = Color.YELLOW;  // 🟡 警告
} else {
    color = Color.RED;     // 🔴 危险
}
```

#### 2. 日志队列管理
```typescript
// 先进先出队列，保持最多 10 条
addBattleLog(message: string) {
    this.battleLogs.unshift(message);  // 新日志添加到开头
    if (this.battleLogs.length > 10) {
        this.battleLogs.pop();  // 移除最旧的日志
    }
    this.refreshLog();
}
```

#### 3. Emoji 表情映射
```typescript
getSkillEmoji(skill: string): string {
    const map = {
        'fire': '🔥', 'water': '💧', 'wind': '🌪️',
        'earth': '🪨', 'light': '✨', 'dark': '🌑',
        'attack': '⚔️', 'defend': '🛡️', 'dodge': '💨',
        'crit': '💥', 'victory': '🏆', 'defeat': '💀'
    };
    return map[skill] || '⚔️';
}
```
