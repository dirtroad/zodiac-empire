### 4.1 前端实现

```typescript
// 战斗界面管理器
export class BattleUI {
    // 更新血量条
    updateHPBar(current: number, max: number) {
        const percent = current / max;
        const bar = this.hpBarNode;
        
        // 根据血量改变颜色
        if (percent > 0.5) {
            bar.color = Color.GREEN; // 🟢
        } else if (percent > 0.3) {
            bar.color = Color.YELLOW; // 🟡
        } else {
            bar.color = Color.RED; // 🔴
        }
        
        // 更新进度
        bar.scaleX = percent;
    }
    
    // 显示伤害数字
    showDamage(value: number, isCrit: boolean, target: Node) {
        const label = this.createDamageLabel(value);
        label.string = isCrit ? `💥${value}` : `-${value}`;
        label.color = isCrit ? Color.RED : Color.WHITE;
        
        // 飘字动画
        this.animateDamage(label, target);
    }
    
    // 显示战斗日志
    addBattleLog(message: string) {
        this.logList.unshift(message);
        if (this.logList.length > 10) {
            this.logList.pop();
        }
        this.refreshLog();
    }
}
```

### 4.2 战斗动画

```typescript
// 简单动画序列
async playAttackAnimation(attacker: Node, defender: Node, skill: string) {
    // 1. 攻击方前摇动画
    attacker.setScale(1.2, 1.2);
    await this.wait(300);
    
    // 2. 冲向对方
    const startPos = attacker.position;
    const endPos = defender.position;
    await this.tween(attacker, startPos, endPos, 200);
    
    // 3. 显示技能表情
    this.showSkillEffect(skill, defender.position);
    
    // 4. 受击方震动
    this.shake(defender, 100);
    
    // 5. 返回原位
    await this.tween(attacker, endPos, startPos, 200);
    
    // 6. 恢复大小
    attacker.setScale(1.0, 1.0);
}

// 表情特效
showSkillEffect(skill: string, pos: Vec3) {
    const emoji = this.getSkillEmoji(skill);
    const node = this.createEmoji(emoji);
    node.setPosition(pos);
    
    // 放大淡出动画
    this.animateEmoji(node);
}

getSkillEmoji(skill: string): string {
    const map = {
        'fire': '🔥',
        'water': '💧',
        'wind': '🌪️',
        'earth': '🪨',
        'light': '✨',
        'dark': '🌑',
        'attack': '⚔️',
        'defend': '🛡️',
    };
    return map[skill] || '⚔️';
}
```

## 五、战斗公式（含运气成分）

```typescript
// 战斗伤害计算
calculateDamage(attacker: BattleUnit, defender: BattleUnit, skill: Skill): number {
    // 基础伤害
    let baseDamage = attacker.attack * skill.damageCoefficient;
    
    // 五行克制系数
    const wuxingCoeff = this.getWuxingCoefficient(attacker.wuxing, defender.wuxing);
    // 相克：1.3，相生：0.9，同行：1.0，被克：0.8
    
    // 星座匹配系数
    const zodiacCoeff = this.getZodiacCoefficient(attacker.zodiac, defender.zodiac);
    // 对宫：1.3，同极性：1.1，其他：1.0
    
    // 幸运系数（80%-120% 随机）
    const luckCoeff = 0.8 + Math.random() * 0.4;
    
    // 暴击判定
    const isCrit = Math.random() < attacker.critRate;
    const critCoeff = isCrit ? 1.5 : 1.0;
    
    // 闪避判定
    const isDodge = Math.random() < defender.dodgeRate;
    if (isDodge) {
        return 0; // 完全闪避
    }
    
    // 最终伤害
    let finalDamage = baseDamage * wuxingCoeff * zodiacCoeff * luckCoeff * critCoeff;
    
    // 防御减伤
    const defenseReduction = defender.defense / (defender.defense + 1000);
    finalDamage = finalDamage * (1 - defenseReduction);
    
    return Math.floor(finalDamage);
}

// 胜率计算（用于战前提示）
calculateWinRate(myPower: number, enemyPower: number): number {
    const baseRate = myPower / (myPower + enemyPower);
    
    // 修正系数
    const correction = 1.0 + (Math.random() - 0.5) * 0.2; // ±10% 随机
    
    return Math.min(0.95, Math.max(0.05, baseRate * correction));
}
```
